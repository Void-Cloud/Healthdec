CREATE TABLE User
(
  Password VARCHAR(100) NOT NULL,
  Email VARCHAR(100) NOT NULL,
  Id INT NOT NULL,
  Height INT NOT NULL,
  PRIMARY KEY (Id)
);

CREATE TABLE Meassure
(
  Value INT NOT NULL,
  Type INT NOT NULL,
  IdM INT NOT NULL,
  Timestamp INT NOT NULL,
  Id INT,
  PRIMARY KEY (IdM),
  FOREIGN KEY (Id) REFERENCES User(Id)
);
